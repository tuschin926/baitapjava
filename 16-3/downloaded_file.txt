var showTime = function(unixtime){
    var onNow = new Date();
    var articleTime = new Date(unixtime*1000);
    var ago = parseInt((onNow.getTime() - articleTime.getTime())/1000);
    var minutes = ago/60;
    var hours = ago/3600;
    if(minutes < 60)
    {
        return '<span class="time-ago" datetime="'+articleTime.getTime()+'">'+Math.ceil(minutes)+"' trước</span>";
    }
    else if(minutes >= 60 && hours <= 24)
    {
        return '<span class="time-ago" datetime="'+articleTime.getTime()+'">'+Math.round(hours)+"h trước</span>";
    }
    else if(hours > 24)
    {
        var currentYear = new Date().getFullYear();
        var articleYear = articleTime.getFullYear();
        if(articleYear === currentYear)
        {
            return articleTime.getDate()+'/'+(articleTime.getMonth()+1);
        }
        else if(currentYear > articleYear)
        {
            return articleTime.getDate()+'/'+(articleTime.getMonth()+1)+'/'+articleTime.getFullYear();
        }
    }
}

function setCookie(name,value,days) {
    var expires = "";
    if (days) {
        var date = new Date();
        date.setTime(date.getTime() + (days*24*60*60*1000));
        expires = "; expires=" + date.toUTCString();
    }
    document.cookie = name + "=" + (value || "")  + expires + "; path=/";
}
function getCookie(name) {
    var nameEQ = name + "=";
    var ca = document.cookie.split(';');
    for(var i=0;i < ca.length;i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
    }
    return null;
}
function clearCache(url){
    fetch(url, {
        headers: {
            'Cache-Control': 'no-store'
        }
    }).then(function(res) {
        return res.text();
    }).catch(function(error) {
        console.warn('Failed: ', error);
    }).finally(function(){
        fetch(url, {
            headers: {
                'Cache-Control': 'no-cache'
            }
        }).then(function(res) {
            return res.text();
        }).catch(function(error) {
            console.warn('Failed: ', error);
        });
    });
}
